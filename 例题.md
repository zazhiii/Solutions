import java.io.*;
import java.util.HashSet;
import java.util.Scanner;
public class Main {
	static HashSet<Integer> hs = new HashSet<>();
	static int[] arr = new int[1000000];
	static boolean[] used = new boolean[1000000];
	static boolean[] have = new boolean[1000000];
	public static void main(String[] args) throws IOException {
       //读数据
	   Scanner s = new Scanner(System.in);
	   int n,i=0;
	   while(true) {
		   n = s.nextInt();
		   if(n==0)break;
		   arr[i++] = n;
	   }
	   //
	   
	   for (int j = 0; j <=i-1; j++) {
		dfs(0, 0, arr[j],j);
		if(have[j]==false) {
			System.out.println("There are no solutions for "+arr[j]+".");
		}
	}

	   
	}
	public static void dfs(int num, int cur,int n,int k) {//num之前选的数组成的数字，cur当前选了几个数了
		if (cur==5) {										//k:判断第k个数
			int num2 = n*num;
			if (ok(num2)) {
				have[k]= true; //第k个数有解 
				System.out.println(num2+"/"+String.format("%05d", num)+"="+n);
			}
			return;
		}

		for (int i = 0; i <=9; i++) {//从0~9开始选 这样包含了所有排列
			if (used[i]==true) {
				continue;
			}
			num = num*10+i;
			used[i] = true;
			hs.add(i);
			dfs(num, cur+1,n,k);
			num = (num-i)/10;
			used[i] = false;
			hs.remove(i);
		}
	}
	private static boolean ok(int num2) {
		int n = num2;
		if (num2>98765) {
			return false;
		}
		HashSet<Integer> hs2 = new HashSet<>();
		for(int i = 1;i<=5;i++) {
			int x = n%10;
			if (hs.contains(x)||hs2.contains(x)) {//与另一个数字有重合||与自身有
				return false;
			}
			hs2.add(x);
			n = n/10;
		}
		return true;
	}			
}
